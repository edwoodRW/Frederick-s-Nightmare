#AoNF loses Rothermere's support

	id = britain.5
	title = britain.5.t
	desc = britain.5.d
	picture = GFX_report_event_fascist_gathering
	
	is_triggered_only = yes
	
	trigger = {
		NOT = {
			has_government = fascism

	}

		name = britain.5.o1

	}
}
